1   package nimirum.Miinaharavainen.kuuntelijat;
2   
3   import java.awt.event.MouseEvent;
4   import java.util.ArrayList;
5   import javax.swing.SwingUtilities;
6   import javax.swing.event.MouseInputAdapter;
7   import nimirum.Miinaharavainen.gui.Piirtaja;
8   import nimirum.Miinaharavainen.gui.TapahtumaAlue;
9   
10  /**
11   * Kuuntelee käyttäjän hiirenklikkauksia tapahtumaAlueiden päällä
12   *
13   * @author nimirum
14   */
15  public class KlikkaustenKuuntelija extends MouseInputAdapter {
16  
17      private final Piirtaja piirtaja;
18      private final ArrayList<TapahtumaAlue> tapahtumaalueet;
19  
20      /**
21       * Konstruktori
22       *
23       * @param piirtaja
24       * @param tapahtumaAlueet
25       */
26      public KlikkaustenKuuntelija(Piirtaja piirtaja, ArrayList tapahtumaAlueet) {
27          this.piirtaja = piirtaja;
28          tapahtumaalueet = tapahtumaAlueet;
29      }
30  
31      @Override
32      public void mouseClicked(MouseEvent e) {
33          if (SwingUtilities.isLeftMouseButton(e)) {
34              for (TapahtumaAlue alue : tapahtumaalueet) {
35                  alue.alueeseenKlikattu(e.getX(), e.getY());
36              }
37          }
38          if (SwingUtilities.isRightMouseButton(e)) {
39              for (TapahtumaAlue alue : tapahtumaalueet) {
40                  alue.alueenLiputus(e.getX(), e.getY());
41              }
42          }
43          piirtaja.repaint();
44      }
45  
46  }
