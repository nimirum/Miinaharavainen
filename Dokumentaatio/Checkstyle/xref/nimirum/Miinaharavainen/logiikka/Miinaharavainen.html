1   package nimirum.Miinaharavainen.logiikka;
2   
3   import nimirum.Miinaharavainen.gui.Kayttoliittyma;
4   import nimirum.Miinaharavainen.highscore.EnnatystenKasittelija;
5   
6   /**
7    * Miinharava peli luokka, joka luo pelilaudan
8    *
9    * @author nimirum
10   */
11  public final class Miinaharavainen {
12  
13      private final Pelilauta pelilauta;
14      private int miinojenMaara = 0;
15      private boolean miinoitettu = false;
16      private final Kello pelikello;
17      private String gameOver = null;
18      private final EnnatystenKasittelija kasittelija;
19  
20      /**
21       * Miinaharavan konstruktori, missä luodaan pelilauta annetuista arvoista.
22       * Pelilaudan mittojen pitää olla väliltä 8-40, muuten luo 10x10 pelilaudan.
23       *
24       * @param leveys Pelilaudan leveys
25       * @param korkeus Pelilaudan korkeus
26       */
27      public Miinaharavainen(int leveys, int korkeus) {
28          if (8 <= leveys && leveys <= 40 && 8 <= korkeus && korkeus <= 40) {
29              pelilauta = new Pelilauta(leveys, korkeus);
30          } else {
31              pelilauta = new Pelilauta(10, 10);
32          }
33          pelikello = new Kello();
34          kasittelija = new EnnatystenKasittelija();
35      }
36  
37      /**
38       * Miinoittaa pelilaudan random lukujen avulla ensimmäisen klikkauksen
39       * jälkeen. Asetttaa miinojen määrän muistiin, jota tarvitsee pelin
40       * voittamiseen. Ajanoton pitäisi alkaa tämän metodin käynnistämisen jälkeen
41       *
42       * @param klikkausX Ensimmäisen klikkauksen x koordinaatti
43       * @param klikkausY Ensimmäisen klikkauksen y koordinaatti
44       */
45      public void miinoitaLauta(int klikkausX, int klikkausY) {
46          if (!miinoitettu) {
47              pelilauta.miinoita(klikkausX, klikkausY);
48              miinojenMaara = pelilauta.getMiinojenMaara();
49              miinoitettu = true;
50              pelikello.run();
51          }
52      }
53  
54      public Kello getKello() {
55          return pelikello;
56      }
57  
58      public boolean isPelilautaMiinoitettu() {
59          return miinoitettu;
60      }
61  
62      public int getMiinojenMaara() {
63          return miinojenMaara;
64      }
65  
66      public Pelilauta getPelilauta() {
67          return pelilauta;
68      }
69  
70      /**
71       * Pelin päättyminen, kello pysähtyy, kentän koko ja aika tallennetaan
72       * ennätyksiin jos kyseessä voittaminen
73       *
74       * @param tapaus Voitto tai häviö
75       */
76      public void gameOver(String tapaus) {
77          gameOver = tapaus;
78          pelikello.stop();
79          getPelilauta().avaaKaikkiRuudut();
80      }
81  
82      public EnnatystenKasittelija getKasittelija() {
83          return kasittelija;
84      }
85  
86      public String getGameOver() {
87          return gameOver;
88      }
89  }
