3   import java.util.ArrayList;
4   import nimirum.Miinaharavainen.logiikka.Miinaharavainen;
5   
6   /**
7    * Luo Tapahtumalueet ja niiden koordinaatit
8    *
9    * @author nimirum
10   */
11  public class Sijainnit {
12  
13      private final Kayttoliittyma kayttoliittyma;
14      private final Miinaharavainen harava;
15      private final int ruudunLeveys;
16      private final int ruudunKorkeus;
17  
18      /**
19       * Konstruktori
20       *
21       * @param miinaharava
22       * @param kayttoliittyma
23       */
24      public Sijainnit(Miinaharavainen miinaharava, Kayttoliittyma kayttoliittyma) {
25          this.harava = miinaharava;
26          this.kayttoliittyma = kayttoliittyma;
27          ruudunLeveys = miinaharava.getPelilauta().getRuutu(0, 0).getRuudunLeveys();
28          ruudunKorkeus = miinaharava.getPelilauta().getRuutu(0, 0).getRuudunLeveys();
29      }
30  
31      /**
32       *
33       * @return Lista tapahtuma-alueista
34       */
35      public ArrayList tapahtumaAlueet() {
36          ArrayList list = new ArrayList();
37          for (int i = 0; i < harava.getPelilauta().getX() * ruudunLeveys; i = i + ruudunLeveys) {
38              for (int j = 0; j < harava.getPelilauta().getY() * ruudunKorkeus; j = j + ruudunKorkeus) {
39                  TapahtumaAlue alue = new TapahtumaAlue(i, j, harava.getPelilauta().getRuutu(i / ruudunLeveys, j / ruudunKorkeus), harava, kayttoliittyma);
40                  list.add(alue);
41              }
42          }
43          return list;
44      }
45  
46  }
